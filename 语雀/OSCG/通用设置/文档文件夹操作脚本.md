# 文档文件夹操作脚本

```
for doc in records:
    # 解析文件名以获取产品代码
    doc_name, ext = doc.name.rsplit(".", 1) if "." in doc.name else (doc.name, "")
    product_code = doc_name.split("_")[-1].strip() if "_" in doc_name else doc_name.strip()
    product_tmpl = env["product.template"].search([("default_code", "=", product_code)], limit=1)

    # 更新产品附件
    if product_tmpl:
 
        # 获取标准化后的当前文档名
        normalized_name = doc_name.strip().upper()+ext.strip().upper()
        # 搜索所有相关附件
        existing_attachments = env['ir.attachment'].search([
            ('res_model', '=', 'product.template'),
            ('res_id', '=', product_tmpl.id)
        ])
        # 比较文件名是否一致，如果一致，则删除
        for attachment in existing_attachments:
            doc_name1, ext1 = attachment.name.rsplit(".", 1) if "." in attachment.name else (attachment.name, "")
            normalized_name1 = doc_name1.strip().upper()+ext1.strip().upper()
            if normalized_name1 == normalized_name:
                attachment.unlink()
            
        # 删除同名文件后，把刚上传的文件标识上产品id
        doc.write({"res_model":"product.template", "res_id":product_tmpl.id})
        
        attachment = env['ir.attachment'].search([
            ('res_model', '=', 'product.template'),
            ('res_id', '=', product_tmpl.id),
            ('mimetype', '=', 'application/pdf'),
            ('name', 'ilike', '图纸文件')
        ], limit=1)
        
        if attachment:
            # 查找所有包含该产品的BOM
            boms = env['mrp.bom'].search([('product_tmpl_id', '=', product_tmpl.id)])
            
            # 遍历每个BOM，找到对应的工艺路线，并更新相关工序
            for bom in boms:
                workcenters = bom.operation_ids
                if workcenters:
                  
                    for workcenter in workcenters:
                        workcenter.write({
                            'worksheet_type': 'pdf',
                            'worksheet': attachment.datas,
                        })
                        
        doc.write({'folder_id': 26340})
    # else:
    #     log('没有找到相应的物料代码'+product_code, level='info')
```
