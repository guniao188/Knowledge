3.复制之后，执行脚本

```
   #其中x_studio_是在页面上加的层级

for rec in record:  

    if not rec.page_id or not rec.page_id.x_studio_:
        continue
    code = rec.page_id.x_studio_.strip()

    # 跳过没有 '.' 的顶级节点（它们没有父级）
    if '.' not in code:
        continue

    # 计算期望的父级 code：去掉最后一段
    expected_parent_code = code.rsplit('.', 1)[0]

    # 查找所有同网站的菜单
    if not rec.website_id:
        continue

    menu_ids = rec.env['website.menu'].search([
        ('website_id', '=', rec.website_id.id),
        ('page_id', '!=', False)  # 确保有 page_id
    ])

    parent_menu = None
    for menu in menu_ids:
        origin = menu.page_id.x_studio_ or ''
        if origin.strip() == expected_parent_code:
            parent_menu = menu
            break

    # 设置 parent_id（假设你的模型有 parent_id 字段指向 website.menu）
    parent_id = parent_menu.id if parent_menu else False
    rec.write({'parent_id':parent_id})
```
