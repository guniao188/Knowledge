```
@api.model
def create(cls, values):
    self = super().create(values)
    self.set_code()
    return self

def set_code(self):
    if not self.product_scenario_id:
        raise UserError('')

    all_selfs = self.search([('')],)
    max_name = max(all_selfs.mapped('name'))
    # int(01)=>1+1=>2=>'2'=>'02'
    # 11=>011=>
    last_sequence = ('0'+str(int(max_name[-2:])+1))[-2:]
    self.code = ''.join([self.product_scenario_id.code, self.product_material.code, last_sequence])
```
