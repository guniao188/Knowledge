# 下料导入

```
<?xml version="1.0" encoding="utf-8"?>
<odoo><data>

        <record id="demo_batch_import_server_action" model="ir.actions.server">
            <field name="name">下料消耗原批次，创建新批次</field>
            <field name="model_id" ref="base_server_action_params.model_base_action_param_wizard"/>
            <field name="state">code</field>
            <field name="code">

def _quant_import(env, excel_data):
    # 初始化一个字典来存储每种产品的最新库存数量
    current_quantities = {}
    NewQuant, OriginQuant = env['stock.quant'], env['stock.quant']
    for line in excel_data:
        pmaterial = str(line["材质"]).strip()
        psize = str(line["规格"]).strip()
        if str(line["规格"]) != line["规格"] and psize.endswith('.0'):
            psize = psize[:-2]
        product_name = str(pmaterial) + '-' + str(psize)
        plot_name = str(line["板材编号"]).strip()
        if str(line["板材编号"]) != line["板材编号"] and plot_name.endswith('.0'):
            plot_name = plot_name[:-2]
        p_new_lot_name = str(line["零件号"]).strip()
        pnumber = float(str(line["数量"]).strip())
        per_weight = float(str(line["单重"]).strip())
        product_num = pnumber * per_weight

        # 查找产品
        product = env['product.product'].search([('name', '=', product_name)], limit=1)
        if not product:
            raise UserError(f"未找到产品: {product_name}")

        # 查找现有的批次
        lot = env['stock.lot'].search([('name', '=', plot_name)], limit=1)
        if not lot:
            raise UserError(f"未找到批次: {plot_name}")

        # 查找现有的库存记录
        quants = env['stock.quant'].search([
            ('product_id', '=', product.id),
            ('lot_id', '=', lot.id),
            ('location_id.usage', '=', 'internal')
        ])
        OriginQuant |= quants

        # 获取当前库存数量
        if (product.id, lot.id) not  in current_quantities:
            current_quantity = current_quantities[(product.id, lot.id)]
        elif quants:
            current_quantity = sum(quants.mapped('quantity'))
        else:
            raise UserError(f"未找到符合条件的库存记录。产品名称: {product_name}, 板材编号: {plot_name}")

        # 创建新批次
        new_lot = env['stock.lot'].create({
            'name': p_new_lot_name,
            'product_id': product.id,
            'company_id': env.company.id  # 假设当前公司是默认公司
        })

        # 更新库存数量
        new_quantity = current_quantity - product_num
        if new_quantity < 0:
            raise UserError(f"库存不足，无法完成操作。产品名称: {product_name}, 板材编号: {plot_name}")

        # 更新库存记录
        if quants:
            quants[0].write({'inventory_quantity': new_quantity})
        else:
            raise UserError(f"未找到符合条件的库存记录。产品名称: {product_name}, 板材编号: {plot_name}")

        # 插入新批次的库存记录
        new_quant = env['stock.quant'].create({
            'product_id': product.id,
            'lot_id': new_lot.id,
            'quantity': 0.0,
            'location_id': quants[0].location_id.id  # 假设使用相同的库存位置
        })
        NewQuant |= new_quant

        # 更新 inventory_quantity
        new_quant.write({'inventory_quantity': product_num})

        # 更新当前库存数量
        current_quantities[(product.id, lot.id)] = new_quantity
    env['stock.inventory.adjustment.name'].create({
        'quant_ids': [(6,0,(NewQuant | OriginQuant).ids)],
        'inventory_adjustment_name': '裁切下料'
    }).action_apply()

# 处理导入的Excel文件
if excel_data:
    _quant_import(env, excel_data)

            </field>
        </record>

</data></odoo>
```
