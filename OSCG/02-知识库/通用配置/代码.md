    @api.depends('product_uom_qty', 'move_ids.state')
    def _compute_qty_to_receive(selfs):
        for self in selfs:

                self.product_id.type != 'service',
                self.order_id.state == 'purchase',
            ]):

@api.depends('move_ids.state')
    def _compute_latest_done_stage(selfs):
        for self in selfs:
            moves = self.move_ids
            while 'done' not in set(moves.mapped('state')) and moves:
                moves = moves.move_orig_ids
            if moves:
                latest_done_stage = moves[:1].picking_type_id.name
            else:
                latest_done_stage = ''
            self.latest_done_stage = latest_done_stage
            
latest_done_stage = fields.Char('完成工序', compute='_compute_latest_done_stage')

 @api.model
    def _name_search(cls, name, args=None, operator='ilike', limit=100, name_get_uid=None):
        result = super()._name_search(name, args=args, operator=operator, limit=limit, name_get_uid=name_get_uid)
        super_set = set(result)
        if cls._context.get('show_info'):
            try:
                _digit = float(name)
            except ValueError:
                pass
            else:
                this_set = set(
                    cls.search(['|', ('product_width', '=', _digit),
                                ('product_length', '=', _digit), ], limit=limit).ids
                )
                super_set.update(this_set)
            
        return list(super_set)[:limit]
